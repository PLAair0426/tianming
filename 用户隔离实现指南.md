@router.post("/divination/generate-line", response_model=SingleLineResponse)
async def generate_single_line(req: Request):
    """
    生成单个爻
    
    使用蓍草法生成一个爻的数值（6,7,8,9）
    所有卦象数据都由后端processor生成，确保数据一致性
    
    注意：生成单个爻时不消耗元气，元气消耗在完成6个爻后统一结算。

        # 获取用户IP
        client_ip = req.client.host
        
        # 获取该用户的元气系统
        karma_system = user_karma_systems[client_ip]
        
        # 直接生成爻，不消耗元气

        name_map = {6: "老阴", 7: "少阳", 8: "少阴", 9: "老阳"}

        # 返回当前元气状态（不消耗）
        karma_status = karma_system.get_status()
        
        return SingleLineResponse(
            value=value,
            binary=binary,
            name=name,
            karma_status=karma_status
        )
    except Exception as e:

**修改为**：
```python
@router.post("/divination/interpret", response_model=DivinationResponse)
async def interpret_divination(request: DivinationRequest, req: Request):
    """
    AI解读占卜结果
    
    接收用户问题和卦象数据，返回AI解读结果
    
    采用三阶段模式结算元气：
    1. 验资（preview）：检查元气是否足够，不扣费
    2. 生成：执行卦象生成和AI解读逻辑
    3. 结算（commit）：生成成功后扣除元气
    
    如果生成失败，不会扣除元气，保证公平交易。
    """
    try:
        # 获取用户IP
        client_ip = req.client.host
        
        # 获取该用户的元气系统
        karma_system = user_karma_systems[client_ip]
        
        # 更新元气值（考虑衰减和恢复）
        karma_system.update_vitality()
        
        # 检查元气是否足够（元气为0时不能使用）
        if karma_system.current_vitality <= 0:
            raise HTTPException(
                status_code=429,
                detail={
                    "error": "元气不足",
                    "message": "元神枯竭，无法窥探天机。请稍作休息，或考虑充值恢复元气。",
                    "current_vitality": karma_system.current_vitality,
                    "required_cost": 0,
                    "karma_status": karma_system.get_status()
                }
            )
        
        # --- 阶段一：验资（Pre-check）---
        preview = karma_system.calculate_cost_preview()
        
        if not preview["can_afford"]:
            raise HTTPException(
                status_code=429,
                detail={
                    "error": "元气不足",
                    "message": "元神枯竭，无法窥探天机",
                    "current_vitality": preview["current_vitality"],
                    "required_cost": preview["estimated_cost"],
                    "karma_status": karma_system.get_status()
                }
            )
        
        # --- 阶段二：生成卦象和AI解读 ---
        # ... 后续代码保持不变，但所有使用 karma_system 的地方都已经使用用户专属的实例
        
        # --- 阶段三：结算扣费 ---
        commit_success = karma_system.commit_transaction(preview["estimated_cost"])
        
        if not commit_success:
            raise HTTPException(
                status_code=500,
                detail="结算失败，请重试"
            )
        
        # 记录使用
        karma_system.record_use()
        
        # ... 返回结果
```

## ⚠️ 注意事项

### 1. IP 地址获取

FastAPI 的 `req.client.host` 可能获取到代理后的IP（如果使用CDN或反向代理）。

**更好的方式**（如果使用 Cloudflare）：
```python
# 获取真实IP（考虑代理）
def get_client_ip(req: Request) -> str:
    # 优先使用 X-Forwarded-For 头（如果有代理）
    forwarded_for = req.headers.get("X-Forwarded-For")
    if forwarded_for:
        # 取第一个IP（原始客户端IP）
        return forwarded_for.split(",")[0].strip()
    # 否则使用直接连接的IP
    return req.client.host if req.client else "unknown"

# 使用方式
client_ip = get_client_ip(req)
karma_system = user_karma_systems[client_ip]
```

### 2. 内存管理

- `defaultdict` 会自动为新用户创建元气系统
- 如果用户很多，内存占用会增加
- 可以考虑定期清理不活跃用户的元气系统（可选）

### 3. NAT 网络问题

- 同一网络下的用户（如公司、学校）可能共享同一个公网IP
- 这些用户会共享元气系统
- 这是基于IP方案的局限性

### 4. 测试

修改后需要测试：
- 不同IP的用户是否有独立的元气系统
- 同一IP的用户是否共享元气系统
- 元气消耗是否独立
